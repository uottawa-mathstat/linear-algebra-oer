\item $\set{(1,1), (1, 2)}$; $\R^2$
\medskip

\item\sov $\set{(1,1), (2, 2)}$; $\R^2$.

\medskip

\medskip

\medskip
%

\medskip
%
 
\item\sov $\set{(1,1,1), (1,0,3), (0,3,4)}$; $\R^3$.


\medskip 
%
\item $\set{(1,0,0), (0,1,0), (0,0,1), (1,2,3)}$; $\R^3$
\medskip
%
\item\sov $\set{(0,-3), (3, 0)}$; $\R^2$.
